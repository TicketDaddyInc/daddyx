Addition 1: CreatorRegistry.sol
solidity

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.24;

import "@openzeppelin/contracts/access/Ownable.sol";

contract CreatorRegistry is Ownable {
    enum CreatorStatus { None, Pending, Approved, Suspended }

    struct CreatorProfile {
        address wallet;
        string name;
        string country;
        string email;
        uint256 appliedAt;
        uint256 approvedAt;
        uint256 eventCount;
        CreatorStatus status;
    }

    mapping(address => CreatorProfile) public creators;
    address[] public pendingQueue;

    event CreatorApplied(address indexed creator, string name);
    event CreatorApproved(address indexed creator, address indexed approvedBy);
    event CreatorSuspended(address indexed creator);

    modifier onlyApprovedCreator() {
        require(creators[msg.sender].status == CreatorStatus.Approved, "Not an approved DaddyX Creator");
        _;
    }

    function applyAsCreator(
        string calldata name,
        string calldata country,
        string calldata email
    ) external {
        require(creators[msg.sender].status == CreatorStatus.None, "Already applied");
        creators[msg.sender] = CreatorProfile({
            wallet: msg.sender,
            name: name,
            country: country,
            email: email,
            appliedAt: block.timestamp,
            approvedAt: 0,
            eventCount: 0,
            status: CreatorStatus.Pending
        });
        pendingQueue.push(msg.sender);
        emit CreatorApplied(msg.sender, name);
    }

    function approveCreator(address creator) external onlyOwner {
        require(creators[creator].status == CreatorStatus.Pending, "Not pending");
        creators[creator].status = CreatorStatus.Approved;
        creators[creator].approvedAt = block.timestamp;
        emit CreatorApproved(creator, msg.sender);
    }

    function suspendCreator(address creator) external onlyOwner {
        creators[creator].status = CreatorStatus.Suspended;
        emit CreatorSuspended(creator);
    }

    function isApproved(address creator) external view returns (bool) {
        return creators[creator].status == CreatorStatus.Approved;
    }
}
Addition 2: Updated DaddyXCampaign.sol — Full Fund Flow Architecture
The DaddyXCampaign contract is the DaddyX-specific escrow. It is separate from, but integrates with, the existing TicketDaddy Event Escrow.

plaintext

Fund Flow:
------------------------------------------------------------------
TOKEN SALE (pre-event)
  Fan buys token → DaddyXCampaign escrow
  Step-pricing payout → previous token holder (immediate)
  Platform fee → TicketDaddy treasury
  Remainder → DaddyXCampaign escrow (organizer fund pool)

MILESTONE RELEASE (to organizer, during campaign)
  Organizer calls requestMilestone() → admin approves → funds released to organizer

TICKET SALES (during event)
  Existing TicketDaddy Event Escrow holds ticket proceeds

SETTLEMENT (post-event)
  Oracle calls reportRevenue(ticketRevenue) → stored in DaddyXCampaign
  Oracle reads both escrows; total revenue = ticketRevenue (from ticket escrow)
  Final token holders call claimRevenue() → receive:
    ticketRevenue * revenueShareBps / 10000 / tokenCount
  Any unclaimed escrowed capital after 180 days → organizer fallback withdrawal

CANCELLATION
  Admin or organizer calls cancel()
  Current token holders call claimRefund() → receive last purchase price from escrow
------------------------------------------------------------------
