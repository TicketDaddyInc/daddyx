Prisma Schema
prisma

model Event {
  id              String   @id @default(cuid())
  eventConfigPda  String   @unique
  name            String
  venueName       String
  eventDate       DateTime
  organizerWallet String
  revenueShareBps Int
  initialPriceSol Float
  stepFactorBps   Int
  payoutFactorBps Int
  tokenCount      Int
  cancelled       Boolean  @default(false)
  revenueReported Boolean  @default(false)
  settledRevenue  Float?
  createdAt       DateTime @default(now())
  tokens          Token[]
}

model Token {
  id            String   @id @default(cuid())
  event         Event    @relation(fields: [eventId], references: [id])
  eventId       String
  tokenId       Int
  currentOwner  String
  currentPrice  Float
