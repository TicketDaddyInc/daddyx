1.1 Account Structures

#[account]
pub struct EventConfig {
    pub organizer: Pubkey,           // event organizer wallet
    pub event_id: [u8; 32],          // unique event identifier (hash of event name + timestamp)
    pub revenue_share_bps: u64,      // basis points of net revenue for token holders (e.g., 2000 = 20%)
    pub initial_price: u64,          // initial token price in lamports
    pub step_factor_bps: u64,        // [S] in bps (e.g., 15000 = 1.5x, 20000 = 2.0x)
    pub payout_factor_bps: u64,      // [P] in bps (e.g., 12000 = 1.2x → 20% ROI)
    pub token_count: u64,            // total tokens issued for this event
    pub end_date: i64,               // Unix timestamp of event settlement
    pub cancelled: bool,
    pub revenue_reported: bool,
    pub settled_revenue: u64,        // lamports reported by oracle
    pub oracle: Pubkey,              // TicketDaddy oracle wallet address
    pub platform_fee_bps: u64,       // platform fee in bps (e.g., 300 = 3%)
    pub bump: u8,
}
